//! One assistant/model response cycle executor.

use async_trait::async_trait;
use eos_tool_ports::ToolResult;

use super::AgentLoopState;
use crate::EngineError;

/// Result of one assistant turn.
#[allow(dead_code)] // Phase 2 API shell; variants are produced by Phase 5 wiring.
#[derive(Debug, Clone, PartialEq, Eq)]
pub(crate) enum AssistantTurnResult {
    /// Continue the agent loop.
    Continue,
    /// A terminal tool submitted successfully.
    TerminalToolSubmitted {
        /// Terminal tool result.
        outcome: ToolResult,
    },
}

/// Executes one complete assistant/model response cycle.
#[async_trait]
pub(crate) trait AssistantTurnExecutor: Send + Sync {
    /// Execute one assistant turn against mutable loop state.
    async fn execute_assistant_turn(
        &self,
        state: &mut AgentLoopState,
    ) -> Result<AssistantTurnResult, EngineError>;
}

/// Provider/tool implementation placeholder for production composition.
#[allow(dead_code)] // Constructed when runtime composition switches to AgentLoopLauncher.
#[derive(Debug, Default)]
pub(crate) struct ProviderToolAssistantTurnExecutor;

impl ProviderToolAssistantTurnExecutor {
    /// Build the provider/tool assistant-turn executor.
    #[allow(dead_code)] // Called by the runtime composition phase of this migration.
    #[must_use]
    pub(crate) fn new() -> Self {
        Self
    }
}

#[async_trait]
impl AssistantTurnExecutor for ProviderToolAssistantTurnExecutor {
    async fn execute_assistant_turn(
        &self,
        _state: &mut AgentLoopState,
    ) -> Result<AssistantTurnResult, EngineError> {
        Err(EngineError::Internal(
            "provider/tool assistant-turn executor is not wired yet".to_owned(),
        ))
    }
}
